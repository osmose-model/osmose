 */

 * 2. Number of replicates

 */

 * 3. Configuration filename
 * File for main options of the simulation (grid, time, etc.) 
 */
;gob-configuration;
/*
 * 4. Species filename
 * File for species biological parameters
 */	
;gob-species;
/* 
 * 5. Predation filename
 * Size ratios and thresholds 
 */
;gob-predation;
/*
 * 6. Fishing filename
 * File of fishing mortality
 * @see 16. Fishing seasonality filename for setting seasonal patterns
 */
;gob-fishing;
/*
 * 7. Areas filename
 * File of species distributions by age & time step
 */
//;gob-areas-random;
;gob-areas-maps;
/*
 * 8. Plankton groups filename
 */
;ltl/gob-ltlbasis;
/*
 * 9. Plankton data filename
 */    	
;ltl/gob-ltlforcing;
/*
 * 10. Calibration filename
 */
;gob-calibration;
/*
 * 11. Ouput filename
 * File specifying which indicators have to be provided
 */
;gob-output-clbr;
/*
 * 12. Accessibilities filename
 * 'default' means accessibility = 1
 */
;gob-accessibility;
/*
 * 13. Size of age-class zero filename
 * 'default' means egg size = 1mm
 */
;default;
/*
 * 14. Reproduction filename
 * File for reproduction parameters & seasonality
 * 'default' means constant reproduction rate = 1 / number of time-steps per year
 */
;gob-repro;
/*
 * 15. Fishing seasonality filename
 * 'defautl' means uniform fishing all year long
 */
;gob-fishing-seasonality;
/*
 * 16. Migration filename
 * File of migration occurrence out of the zone
 */
;gob-migration;
/*
 * 17. Coastline filename
 * 'default' means no coast in the grid
 */
;gob-coast;
/*
 * 18. MPA filename
 * 'default' means no MPA
 */
;default;
/*
 * 19. Coupling filename
 * File for two-way coupling configuration
 * The parameter is ignored unless forcing = true in the configuration filename
 */
;none;
